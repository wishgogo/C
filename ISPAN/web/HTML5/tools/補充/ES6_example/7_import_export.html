☆ export: 匯出「變數值」、「函數」或「物件」 export 有兩種： - named export（具名匯出）：匯出獨立的「變數值」、「函數」或「物件」，匯出前必須給名稱，匯入時也必須使用相同的名稱。 一個檔案內可以有好幾個具名匯出。 [ex]. 匯出具名的變數值 export let ans = true; export let obj = { ..... }; ============================== 匯出具名函數(方法) export function temp() { console.log(`do nothing~`); } ============================== 匯出具名物件 const num = 5; const objA = { ..... }; const objB = { ..... }; export { num, objA, objB }; - default export：一個檔案只能有一個 default export。 <== 比較常使用 [ex]. 匯出物件 const obj1 = { ..... }; const obj2 = { ..... }; export default { obj1, obj2 }; ============================== 匯出匿名函數 export default function() { console.log(`do nothing~`); } ============================== 匯出 class export default class { // JS 類別的定義，參考5_classes.html constructor(name, age, gender) { this.name = name; this.age = age; this.gender = gender; } displayName(){ console.log(this.name); } } ------------------------------------------------------------------------ ☆ import: 匯入「變數值」、「函數」或「物件」 - 匯入 named export // export export const obj = { ..... }; export function temp() { console.log(`do nothing~`); } // import import { temp } from './new.js'; // 使用解構賦值的語法 { temp } // .js 可省略 temp(); 或 import { temp, obj } from './new.js'; // 使用解構賦值的語法 { temp, obj } // .js 可省略 temp(); alert(obj); - 匯入 default export // export export default function() { console.log(`do nothing~`); } // import 並且給名稱 import theFile from './new.js'; // .js 可省略 theFile();
